% This script is designed to impose the entire book, generating a printer-ready
% PDF specifically for hardcover book production. It efficiently manages
% printer spreads and compensates for page creep.
%
% The book will be printed on A4-sized paper. I have decided that each
% signature will consist of four A4 sheets, resulting in each signature
% effectively containing 16 pages for an A5-sized book.
%
% The book totals 352 pages, including the front and back covers, with 348
% inner pages when these covers are excluded. This leads to a total of 21 and
% 3/4 signatures. The initial 3/4 signature, positioned at the beginning,
% includes 12 pages. Pages E, G, and I, specifically designed in color, may be
% produced separately using a different type of paper. This is followed by 21
% full signatures, each containing 16 pages, summing up to 21 and 3/4
% signatures. This strategic layout simplifies the manufacturing process,
% allowing for separate and specialized production of the color pages.

%
% [Number of pages] / (4 - 1) = [Number of spreads]
% [Number of spreads] x [Thickness of paper] = [Total creep]
%
% For a single signature

\RequirePackage[l2tabu, orthodox]{nag}
\documentclass[openany]{book}
\usepackage[
	paper=landscape,a4paper,% The paper size
	top=0.000000mm,         % Top margin
	bottom=0.000000mm,      % Bottom margin
	inner=0.000000mm,       % Inner margin (the binding edge)
	outer=0.000000mm,       % Outer margin
	headheight=0.000000mm,  % Header height
	headsep=0.000000mm,     % Space between the header and text
	footskip=0.000000mm,    % Space from the baseline of the last line of text
	                        %     to the baseline of the footer
	marginparwidth=0.000mm, % Width of the margin notes
	marginparsep=0.000000mm % Separation between the margin notes and the main
	                        %     text
]{geometry}
